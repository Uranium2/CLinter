Sentence	-> statement

statement	-> IF ( expresison ) { statement }
		|  IF ( expresison ) { statement } ELSE { statement }
		|  SWICH ( expression ) { statement }

void nextTok(int* pos) {
	*pos = *pos + 1;
}
void sentence(Token* tok) {
	int pos = 0;
	statement(token, &pos);
}

void statement(Token* tok, int* pos) {
	if (tok[pos] == "IF") { // int match(Token tok, "IF")
		nextTok(pos);
		OpenPar(tok, pos);
		expression(tok, pos);
		ClosedPar(tok, pos);
		OpenCurl(tok, pos);
		statement(tok, pos);
		ClosedCurl(tok, pos);
		if (tok[pos] == "ELSE") {
			nextTok(pos);
			OpenCurl(tok, pos);
			statement(tok, pos);
			ClosedCurl(tok, pos);
		} else {
			if (tok[pos] != EOF)
				printf("Error\n");
		}
	} else if (tok[pos] == "SWITCH") {
			OpenPar(tok, pos);
			expression(tok, pos);
			ClosedPar(tok, pos);
			// spaces? Return line?
			OpenCurl(tok, pos);
			statement(tok, pos);
			ClosedCurl(tok, pos);
	}
}
